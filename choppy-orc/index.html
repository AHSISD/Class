/** * This function embeds a website into another website using an iframe. * * @param {string} https://mystifying-goodall-147690.netlify.app/ - The URL of the website to be embedded. * @param {string} Choppy-Orc - The ID of the HTML element that will contain the iframe. * @param {number} 300 - The width of the iframe in pixels. * @param {number} 500 - The height of the iframe in pixels. * @returns {void} */ function embedWebsite(url, containerId, width, height) { try { // Get the container element const container = document.getElementById(containerId); // Create the iframe element const iframe = document.createElement('iframe'); iframe.src = url; iframe.width = width; iframe.height = height; // Append the iframe to the container container.appendChild(iframe); } catch (error) { console.error(error); } }
